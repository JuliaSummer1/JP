--1. Вывести всех работников чьи зарплаты есть в базе, вместе с зарплатами.
select * from employees inner join employees_salary on employees.id=employees_salary.id_employee_salary inner join salary on employees_salary.salary_id=salary.id;
--2. Вывести всех работников у которых ЗП меньше 2000.
select * from employees inner join employees_salary on employees.id=employees_salary.id inner join salary on employees_salary.salary_id = salary.id where monthly_salary<'2000';
--3. Вывести все зарплатные позиции, но работник по ним не назначен. (ЗП есть, но не понятно кто её получает.)

--4. Вывести все зарплатные позиции  меньше 2000 но работник по ним не назначен. (ЗП есть, но не понятно кто её получает.)
select * from employees right join employees_salary on employees.id = employees_salary.id_employee_salary join salary on employees_salary.salary_id = salary.id where monthly_salary < '2000';
--5. Найти всех работников кому не начислена ЗП.
